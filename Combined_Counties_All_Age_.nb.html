# 1. Add a County Name Column to Each
centre_age_adjusted <- centre_age_adjusted %>%
  mutate(county = "Centre")

clinton_age_adjusted <- clinton_age_adjusted %>%
  mutate(county = "Clinton")

huntingdon_age_adjusted <- huntingdon_age_adjusted %>%
  mutate(county = "Huntingdon")

juniata_age_adjusted <- juniata_age_adjusted %>%
  mutate(county = "Juniata")

mifflin_age_adjusted <- mifflin_age_adjusted %>%
  mutate(county = "Mifflin")

# 2. Combine All Together
all_counties_age_adjusted <- bind_rows(
  centre_age_adjusted,
  clinton_age_adjusted,
  huntingdon_age_adjusted,
  juniata_age_adjusted,
  mifflin_age_adjusted
)

# 3. View Combined Table
print(all_counties_age_adjusted)

# 4. Plot!

  geom_line(size = 1.2) +

  labs(
    title = "Age-Adjusted Mortality Rates (20182022)",
    subtitle = "Centre, Clinton, Huntingdon, Juniata, and Mifflin Counties",
    x = "Year",
    y = "Age-Adjusted Mortality Rate (per 100,000)",
    color = "County",
    caption = "Source: PA Department of Health, 2023 ACS"
  ) +
  theme_minimal(base_size = 14) +
  theme(
    plot.title = element_text(face = "bold", size = 18, hjust = 0.5),

  )
